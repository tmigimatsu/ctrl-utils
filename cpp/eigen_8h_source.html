55 template<typename Derived>
56 inline typename Eigen::MatrixBase<Derived>::PlainObject
57 PseudoInverse(const Eigen::MatrixBase<Derived>& A, double svd_epsilon = 0,
58  bool* is_singular = nullptr) {
59  unsigned int options = Eigen::MatrixBase<Derived>::ColsAtCompileTime == Eigen::Dynamic ?
60  Eigen::ComputeThinU | Eigen::ComputeThinV :
61  Eigen::ComputeFullU | Eigen::ComputeFullV;
62  Eigen::JacobiSVD<typename Derived::PlainObject> svd(A, options);
63  const auto& S = svd.singularValues();
64  if (svd_epsilon <= 0) {
65  svd_epsilon = std::numeric_limits<typename Derived::Scalar>::epsilon() *
66  std::max(A.cols(), A.cols()) * S(0);
67  }
68  if (is_singular != nullptr) {
69  *is_singular = (S.array() <= svd_epsilon).any();
70  }
71  return svd.matrixV() *
72  (S.array() > svd_epsilon).select(S.array().inverse(), 0).matrix().asDiagonal() *
73  svd.matrixU().adjoint();
74 }

140 Eigen::Matrix<typename Derived::Scalar,3,3>
141 DoubleCrossMatrix(const Eigen::DenseBase<Derived>& x) {
142  EIGEN_STATIC_ASSERT_VECTOR_ONLY(Derived);
143  eigen_assert(x.size() == 3);
144 
145  typedef typename Derived::Scalar Scalar;
146  typename Eigen::internal::nested_eval<Derived,4>::type z(x.derived());
147  Eigen::Matrix<Scalar,3,3> result;
148 
149  const Scalar aa = -z.coeff(0) * z.coeff(0);
150  const Scalar bb = -z.coeff(1) * z.coeff(1);
151  const Scalar cc = -z.coeff(2) * z.coeff(2);
152 
153  result(0,0) = bb + cc;
154  result(1,0) = z.coeff(0) * z.coeff(1);
155  result(2,0) = z.coeff(0) * z.coeff(2);
156 
157  result(0,1) = result(1,0);
158  result(1,1) = aa + cc;
159  result(2,1) = z.coeff(1) * z.coeff(2);
160 
161  result(0,2) = result(2,0);
162  result(1,2) = result(2,1);
163  result(2,2) = aa + bb;
